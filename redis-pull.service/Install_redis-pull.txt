1. tạo file trong thư mục /usr/bin/

vi /usr/bin/redis-pull

2. Copy code trong file "source_redis-pull.txt" dán vào file

3. cấp quyền thực thi cho file

sudo chmod +x /usr/bin/redis-pull

4. tạo file trong thư mục  /lib/systemd/system/

vi /lib/systemd/system/redis-pull.service

5. dán đoạn dưới đây và lưu lại

[Unit]
Description=Redis-pull - Created by ThienNQ
[Service]
Type=simple
ExecStart=/bin/bash redis-pull
Restart=always
RestartSec=3
[Install]
WantedBy=multi-user.target


6. copy nó tới /etc/systemd/system và cấp quyền

cp /lib/systemd/system/redis-pull.service /etc/systemd/system/

sudo chmod 644 /etc/systemd/system/redis-pull.service

7. khởi động dịch vụ và một số lệnh liên quan dến dịch vụ

sudo systemctl daemon-reload

sudo systemctl start redis-pull.service

sudo systemctl status redis-pull.service

sudo systemctl restart redis-pull.service

sudo systemctl stop redis-pull.service
